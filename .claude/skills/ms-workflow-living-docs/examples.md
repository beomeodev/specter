## Example 3: TAG Chain Validation Report

**Scenario**: Validate TAG chain completeness across project

```python
import subprocess
import re
from collections import defaultdict

def validate_tag_chains(cwd: str) -> dict:
    """Validates TAG chain completeness."""

    # Scan all TAGs
    result = subprocess.run(
        ["rg", "@(SPEC|TEST|CODE|DOC):([A-Z]+-[0-9]+)", "-or", "$1:$2"],
        capture_output=True,
        text=True,
        cwd=cwd,
    )

    # Group by TAG ID
    tag_chains = defaultdict(lambda: {"SPEC": 0, "TEST": 0, "CODE": 0, "DOC": 0})

    for line in result.stdout.strip().split("\n"):
        if ":" in line:
            tag_type, tag_id = line.split(":")
            tag_chains[tag_id][tag_type] += 1

    # Validate completeness
    complete_chains = []
    broken_chains = []

    for tag_id, chain in tag_chains.items():
        # Complete chain: SPEC + TEST + CODE (DOC optional)
        if chain["SPEC"] > 0 and chain["TEST"] > 0 and chain["CODE"] > 0:
            complete_chains.append(tag_id)
        else:
            missing = [k for k, v in chain.items() if v == 0 and k != "DOC"]
            broken_chains.append({
                "id": tag_id,
                "chain": dict(chain),
                "missing": missing,
            })

    total = len(tag_chains)
    integrity_score = (len(complete_chains) / total * 100) if total > 0 else 0

    return {
        "total_chains": total,
        "complete_chains": len(complete_chains),
        "broken_chains": len(broken_chains),
        "integrity_score": round(integrity_score, 2),
        "complete": complete_chains,
        "broken": broken_chains,
    }

# Run validation
report = validate_tag_chains("/workspace/specter")
print(f"TAG Chain Integrity: {report['integrity_score']}%")
print(f"Complete chains: {report['complete_chains']}/{report['total_chains']}")

for broken in report['broken']:
    print(f"❌ {broken['id']}: Missing {broken['missing']}")
```

## Example 4: README Sync with Project Status

**Scenario**: Update README with current implementation status

```python
def sync_readme(cwd: str, major_changes: bool = True) -> str:
    """Updates README with project status."""

    if not major_changes:
        return None  # Skip for minor changes

    # Get TAG chain status
    validation = validate_tag_chains(cwd)

    # Get test coverage
    coverage_result = subprocess.run(
        ["pytest", "--cov=src", "--cov-report=term-missing"],
        capture_output=True,
        text=True,
        cwd=cwd,
    )
    coverage_match = re.search(r"TOTAL.*?(\d+)%", coverage_result.stdout)
    coverage = coverage_match.group(1) if coverage_match else "N/A"

    # Build status section
    status = f"""
## Project Status

- **TAG Chain Integrity**: {validation['integrity_score']}%
- **Complete Chains**: {validation['complete_chains']}/{validation['total_chains']}
- **Test Coverage**: {coverage}%
- **Tests**: All passing ✅
- **Last Updated**: {date.today().strftime("%Y-%m-%d")}

### Feature Implementation

| Feature | Status | TAG Chain |
|---------|--------|-----------|
"""

    # Add feature rows (simplified)
    for tag_id in validation['complete']:
        status += f"| {tag_id} | ✅ Complete | @SPEC → @TEST → @CODE |\n"

    for broken in validation['broken']:
        missing = ', '.join(broken['missing'])
        status += f"| {broken['id']} | ⚠️ Incomplete | Missing: {missing} |\n"

    # Read current README
    with open(f"{cwd}/README.md", "r") as f:
        readme = f.read()

    # Replace status section (between markers)
    start_marker = "<!-- PROJECT_STATUS_START -->"
    end_marker = "<!-- PROJECT_STATUS_END -->"

    if start_marker in readme and end_marker in readme:
        before = readme.split(start_marker)[0]
        after = readme.split(end_marker)[1]
        updated_readme = f"{before}{start_marker}\n{status}\n{end_marker}{after}"
    else:
        # Append if markers don't exist
        updated_readme = readme + f"\n{start_marker}\n{status}\n{end_marker}\n"

    # Write updated README
    with open(f"{cwd}/README.md", "w") as f:
        f.write(updated_readme)

    return updated_readme

# Execute sync
```

```python
        capture_output=True,
        text=True,
        cwd=cwd,
    ).stdout.strip()

    # Read code
    with open(f"{cwd}/{code_file}", "r") as f:
        content = f.read()

    # Extract functions (simplified)
    functions = extract_functions(content)

    # Generate markdown
    doc = f"# @DOC:{tag_id}\n\n"
    for func in functions:
        doc += f"## {func['name']}\n{func['signature']}\n\n"

    # Write doc
    with open(f"{cwd}/docs/api/{tag_id}.md", "w") as f:

    return (tag_id, duration)

# Sequential execution (slow)
tag_ids = ["AUTH-001", "AUTH-002", "HOOKS-001", "HOOKS-002", "SKILLS-001"]

start = time.time()
```

## Example 6: Staged Changes Only (Incremental Sync)

**Scenario**: Sync docs only for staged files

```bash
# Stage specific files
git add src/auth/service.py tests/unit/test_auth.py

# Run /ms.up-docs (default: staged changes only)
/ms.up-docs
```

**Python implementation**:
```python
def sync_staged_changes_only(cwd: str) -> dict:
    """Syncs docs only for staged files."""

    # Get staged files
    staged_files = subprocess.run(
        ["git", "diff", "--cached", "--name-only"],
        capture_output=True,
        text=True,
        cwd=cwd,
    ).stdout.strip().split("\n")

    print(f"Staged files: {staged_files}")
    # ['src/auth/service.py', 'tests/unit/test_auth.py']

    # Extract TAGs from staged files
    tag_ids = set()
    for file in staged_files:
        result = subprocess.run(
            ["rg", "@(CODE|TEST):([A-Z]+-[0-9]+)", "-or", "$2", file],
            capture_output=True,
            text=True,
            cwd=cwd,
        )
        if result.stdout:
            tag_ids.update(result.stdout.strip().split("\n"))

    print(f"Affected TAGs: {tag_ids}")
    # {'AUTH-001'}

    # Generate docs only for affected TAGs
    files_updated = []
    for tag_id in tag_ids:
        doc_file = f"docs/api/{tag_id}.md"
        generate_api_doc_for_tag(tag_id, cwd)
        files_updated.append(doc_file)

    return {
        "files_updated": files_updated,
        "staged_files": staged_files,
        "affected_tags": list(tag_ids),
    }

# Execute
result = sync_staged_changes_only("/workspace/specter")
print(result)
```

**Output**:
```json
{
  "files_updated": ["docs/api/AUTH-001.md"],
  "staged_files": ["src/auth/service.py", "tests/unit/test_auth.py"],
  "affected_tags": ["AUTH-001"]
}
```

**Scenario**: Doc sync fails, but commit proceeds

```python
def sync_with_failopen(cwd: str) -> dict:
    """Syncs docs with fail-open error handling."""
    try:
        # Attempt full sync
        result = sync_all_docs(cwd)
        result["status"] = "SUCCESS"
        return result

    except FileNotFoundError as e:
        # Missing files (e.g., spec.md)
        logger.warning(f"Doc sync failed: {e}")
        return {
            "status": "PARTIAL",
            "error": f"Missing file: {e}",
            "files_updated": [],
            "tag_integrity": None,
        }

    except subprocess.CalledProcessError as e:
        # Git command failed
        logger.error(f"Git error: {e}")
        return {
            "status": "FAILED",
            "error": f"Git error: {e.stderr}",
            "files_updated": [],
        }

    except Exception as e:
        # Unknown error
        logger.error(f"Unexpected error: {e}")
        return {
            "status": "FAILED",
            "error": str(e),
            "files_updated": [],
        }

# Execute with fail-open
result = sync_with_failopen("/workspace/specter")

if result["status"] == "SUCCESS":
    print("✅ Doc sync complete")
elif result["status"] == "PARTIAL":
    print(f"⚠️ Partial sync: {result['error']}")
    print("Commit will proceed anyway (fail-open)")
else:
    print(f"❌ Doc sync failed: {result['error']}")
    print("Commit will proceed anyway (fail-open)")

# Proceed with commit regardless of sync status
subprocess.run(["git", "commit", "-m", "feat: implement feature"])
```

## Example 10: Performance Benchmarking

**Scenario**: Measure doc sync performance for different strategies

```python
import time

def benchmark_sync_strategies(cwd: str, tag_count: int = 20):
    """Benchmarks different sync strategies."""

    tag_ids = [f"TAG-{i:03d}" for i in range(1, tag_count + 1)]

    # Strategy 1: Sequential (baseline)
    start = time.time()
    for tag_id in tag_ids:
        generate_api_doc_for_tag(tag_id, cwd)
    sequential_time = time.time() - start

    # Strategy 2: Parallel (4 workers)
    start = time.time()
    with ThreadPoolExecutor(max_workers=4) as executor:
        list(executor.map(lambda tid: generate_api_doc_for_tag(tid, cwd), tag_ids))
    parallel_time = time.time() - start

    # Strategy 3: Incremental (staged changes only)
    staged_tag_ids = tag_ids[:5]  # Only 5 TAGs changed
    start = time.time()
    for tag_id in staged_tag_ids:
        generate_api_doc_for_tag(tag_id, cwd)
    incremental_time = time.time() - start

    # Results
    return {
        "tag_count": tag_count,
        "sequential": round(sequential_time, 2),
        "parallel": round(parallel_time, 2),
        "incremental": round(incremental_time, 2),
        "speedup_parallel": round(sequential_time / parallel_time, 2),
        "speedup_incremental": round(sequential_time / incremental_time, 2),
    }

# Run benchmark
results = benchmark_sync_strategies("/workspace/specter", tag_count=20)
print(results)
```

**Output**:
